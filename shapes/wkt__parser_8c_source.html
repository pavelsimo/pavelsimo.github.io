100 /* Next are the tables used to determine what action to take based on the
101 ** current state and lookahead token. These tables are used to implement
102 ** functions that take a state number and lookahead value and return an
103 ** action integer.
104 **
105 ** Suppose the action integer is N. Then the action is determined as
106 ** follows
107 **
108 ** 0 <= N < YYNSTATE Shift N. That is, push the lookahead
109 ** token onto the stack and goto state N.
110 **
111 ** YYNSTATE <= N < YYNSTATE+YYNRULE Reduce by rule N-YYNSTATE.
112 **
113 ** N == YYNSTATE+YYNRULE A syntax error has occurred.
114 **
115 ** N == YYNSTATE+YYNRULE+1 The parser accepts its input.
116 **
117 ** N == YYNSTATE+YYNRULE+2 No such action. Denotes unused
118 ** slots in the yy_action[] table.
119 **
120 ** The action table is constructed as a single large table named yy_action[].
121 ** Given state S and lookahead X, the action is computed as
122 **
123 ** yy_action[ yy_shift_ofst[S] + X ]
124 **
125 ** If the index value yy_shift_ofst[S]+X is out of range or if the value
126 ** yy_lookahead[yy_shift_ofst[S]+X] is not equal to X or if yy_shift_ofst[S]
127 ** is equal to YY_SHIFT_USE_DFLT, it means that the action is not in the table
128 ** and that yy_default[S] should be used instead.
129 **
130 ** The formula above is for computing the action when the lookahead is
131 ** a terminal symbol. If the lookahead is a non-terminal (as occurs after
132 ** a reduce action) then the yy_reduce_ofst[] array is used in place of
133 ** the yy_shift_ofst[] array and YY_REDUCE_USE_DFLT is used in place of
134 ** YY_SHIFT_USE_DFLT.
135 **
136 ** The following are the tables generated in this section:
137 **
138 ** yy_action[] A single table containing all actions.
139 ** yy_lookahead[] A table containing the lookahead for each entry in
140 ** yy_action. Used to detect hash collisions.
141 ** yy_shift_ofst[] For each state, the offset into yy_action for
142 ** shifting terminals.
143 ** yy_reduce_ofst[] For each state, the offset into yy_action for
144 ** shifting non-terminals after a reduce.
145 ** yy_default[] Default action for each state.
146 */

487 /*
488 ** Find the appropriate action for a parser given the terminal
489 ** look-ahead token iLookAhead.
490 **
491 ** If the look-ahead token is YYNOCODE, then check to see if the action is
492 ** independent of the look-ahead. If it is, return the action, otherwise
493 ** return YY_NO_ACTION.
494 */
495 static int yy_find_shift_action(
496  yyParser *pParser, /* The parser */
497  YYCODETYPE iLookAhead /* The look-ahead token */
498 ){
499  int i;
500  int stateno = pParser->yystack[pParser->yyidx].stateno;
501 
502  if( stateno>YY_SHIFT_MAX || (i = yy_shift_ofst[stateno])==YY_SHIFT_USE_DFLT ){
503  return yy_default[stateno];
504  }
505  assert( iLookAhead!=YYNOCODE );
506  i += iLookAhead;
507  if( i<0 || i>=YY_SZ_ACTTAB || yy_lookahead[i]!=iLookAhead ){
508  if( iLookAhead>0 ){
509 #ifdef YYFALLBACK
510  YYCODETYPE iFallback; /* Fallback token */
511  if( iLookAhead<sizeof(yyFallback)/sizeof(yyFallback[0])
512  && (iFallback = yyFallback[iLookAhead])!=0 ){
513 #ifndef NDEBUG
514  if( yyTraceFILE ){
515  fprintf(yyTraceFILE, "%sFALLBACK %s => %s\n",
516  yyTracePrompt, yyTokenName[iLookAhead], yyTokenName[iFallback]);
517  }
518 #endif
519  return yy_find_shift_action(pParser, iFallback);
520  }
521 #endif
522 #ifdef YYWILDCARD
523  {
524  int j = i - iLookAhead + YYWILDCARD;
525  if( j>=0 && j<YY_SZ_ACTTAB && yy_lookahead[j]==YYWILDCARD ){
526 #ifndef NDEBUG
527  if( yyTraceFILE ){
528  fprintf(yyTraceFILE, "%sWILDCARD %s => %s\n",

531 #endif /* NDEBUG */

533  }

544 ** Find the appropriate action for a parser given the non-terminal
545 ** look-ahead token iLookAhead.
546 **
547 ** If the look-ahead token is YYNOCODE, then check to see if the action is
548 ** independent of the look-ahead. If it is, return the action, otherwise
549 ** return YY_NO_ACTION.
550 */
551 static int yy_find_reduce_action(
552  int stateno, /* Current state number */
553  YYCODETYPE iLookAhead /* The look-ahead token */
554 ){
555  int i;
556 #ifdef YYERRORSYMBOL
557  if( stateno>YY_REDUCE_MAX ){
558  return yy_default[stateno];
559  }
560 #else
561  assert( stateno<=YY_REDUCE_MAX );
562 #endif
563  i = yy_reduce_ofst[stateno];
564  assert( i!=YY_REDUCE_USE_DFLT );
565  assert( iLookAhead!=YYNOCODE );
566  i += iLookAhead;
567 #ifdef YYERRORSYMBOL
568  if( i<0 || i>=YY_SZ_ACTTAB || yy_lookahead[i]!=iLookAhead ){
569  return yy_default[stateno];
570  }
571 #else
572  assert( i>=0 && i<YY_SZ_ACTTAB );
573  assert( yy_lookahead[i]==iLookAhead );
574 #endif
575  return yy_action[i];
576 }

952 /* The main parser program.
953 ** The first argument is a pointer to a structure obtained from
954 ** "ParseAlloc" which describes the current state of the parser.
955 ** The second argument is the major token number. The third is
956 ** the minor token. The fourth optional argument is whatever the
957 ** user wants (and specified in the grammar) and is available for
958 ** use by the action routines.
959 **
960 ** Inputs:
961 ** <ul>
962 ** <li> A pointer to the parser (an opaque structure.)
963 ** <li> The major token number.
964 ** <li> The minor token number.
965 ** <li> An option argument of a grammar-specified type.
966 ** </ul>
967 **
968 ** Outputs:
969 ** None.
970 */
971 void Parse(
972  void *yyp, /* The parser */
973  int yymajor, /* The major token code number */
974  ParseTOKENTYPE yyminor /* The value for the token */
975  ParseARG_PDECL /* Optional %extra_argument parameter */
976 ){
977  YYMINORTYPE yyminorunion;
978  int yyact; /* The parser action. */
979  int yyendofinput; /* True if we are at the end of input */
980 #ifdef YYERRORSYMBOL
981  int yyerrorhit = 0; /* True if yymajor has invoked an error */
982 #endif
983  yyParser *yypParser; /* The parser */
984 
985  /* (re)initialize the parser, if necessary */
986  yypParser = (yyParser*)yyp;
987  if( yypParser->yyidx<0 ){
988 #if YYSTACKDEPTH<=0
989  if( yypParser->yystksz <=0 ){
990  /*memset(&yyminorunion, 0, sizeof(yyminorunion));*/
991  yyminorunion = yyzerominor;
992  yyStackOverflow(yypParser, &yyminorunion);
993  return;
994  }
995 #endif
996  yypParser->yyidx = 0;
997  yypParser->yyerrcnt = -1;
998  yypParser->yystack[0].stateno = 0;
999  yypParser->yystack[0].major = 0;
1000  }
1001  yyminorunion.yy0 = yyminor;
1002  yyendofinput = (yymajor==0);
1003  ParseARG_STORE;
1004 
1005 #ifndef NDEBUG
1006  if( yyTraceFILE ){
1007  fprintf(yyTraceFILE,"%sInput %s\n",yyTracePrompt,yyTokenName[yymajor]);
1008  }
1009 #endif
1010 
1011  do{
1012  yyact = yy_find_shift_action(yypParser,(YYCODETYPE)yymajor);
1013  if( yyact<YYNSTATE ){
1014  assert( !yyendofinput ); /* Impossible to shift the $ token */
1015  yy_shift(yypParser,yyact,yymajor,&yyminorunion);
1016  yypParser->yyerrcnt--;
1017  yymajor = YYNOCODE;
1018  }else if( yyact < YYNSTATE + YYNRULE ){
1019  yy_reduce(yypParser,yyact-YYNSTATE);
1020  }else{
1021  assert( yyact == YY_ERROR_ACTION );
1022 #ifdef YYERRORSYMBOL
1023  int yymx;
1024 #endif
1025 #ifndef NDEBUG
1026  if( yyTraceFILE ){
1027  fprintf(yyTraceFILE,"%sSyntax Error!\n",yyTracePrompt);
1028  }
1029 #endif
1030 #ifdef YYERRORSYMBOL
1031  /* A syntax error has occurred.
1032  ** The response to an error depends upon whether or not the
1033  ** grammar defines an error token "ERROR".
1034  **
1035  ** This is what we do if the grammar does define ERROR:
1036  **
1037  ** * Call the %syntax_error function.
1038  **
1039  ** * Begin popping the stack until we enter a state where
1040  ** it is legal to shift the error symbol, then shift
1041  ** the error symbol.
1042  **
1043  ** * Set the error count to three.
1044  **
1045  ** * Begin accepting and shifting new tokens. No new error
1046  ** processing will occur until three tokens have been
1047  ** shifted successfully.
1048  **
1049  */
1050  if( yypParser->yyerrcnt<0 ){
1051  yy_syntax_error(yypParser,yymajor,yyminorunion);
1052  }
1053  yymx = yypParser->yystack[yypParser->yyidx].major;
1054  if( yymx==YYERRORSYMBOL || yyerrorhit ){
1055 #ifndef NDEBUG
1056  if( yyTraceFILE ){
1057  fprintf(yyTraceFILE,"%sDiscard input token %s\n",
1058  yyTracePrompt,yyTokenName[yymajor]);
1059  }
1060 #endif
1061  yy_destructor(yypParser, (YYCODETYPE)yymajor,&yyminorunion);
1062  yymajor = YYNOCODE;
1063  }else{
1064  while(
1065  yypParser->yyidx >= 0 &&
1066  yymx != YYERRORSYMBOL &&
1067  (yyact = yy_find_reduce_action(
1068  yypParser->yystack[yypParser->yyidx].stateno,
1069  YYERRORSYMBOL)) >= YYNSTATE
1070  ){
1071  yy_pop_parser_stack(yypParser);
1072  }
1073  if( yypParser->yyidx < 0 || yymajor==0 ){
1074  yy_destructor(yypParser,(YYCODETYPE)yymajor,&yyminorunion);
1075  yy_parse_failed(yypParser);
1076  yymajor = YYNOCODE;
1077  }else if( yymx!=YYERRORSYMBOL ){
1078  YYMINORTYPE u2;
1079  u2.YYERRSYMDT = 0;
1080  yy_shift(yypParser,yyact,YYERRORSYMBOL,&u2);
1081  }
1082  }
1083  yypParser->yyerrcnt = 3;
1084  yyerrorhit = 1;
1085 #elif defined(YYNOERRORRECOVERY)
1086  /* If the YYNOERRORRECOVERY macro is defined, then do not attempt to
1087  ** do any kind of error recovery. Instead, simply invoke the syntax
1088  ** error routine and continue going as if nothing had happened.
