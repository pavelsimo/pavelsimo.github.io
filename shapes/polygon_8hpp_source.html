184  template <typename CoordInputIt, typename OffsetInputIt>
185  Polygon(CoordInputIt coord_first, CoordInputIt coord_last, OffsetInputIt offset_first, OffsetInputIt offset_last, DimensionType input_dim)
186  {
187  dim = input_dim;
188  if (std::distance(coord_first, coord_last) > 0 and std::distance(offset_first, offset_last) > 0)
189  {
190  auto ndim = static_cast<size_t>(utils::get_ndim(input_dim));
191 
193  size_t lo = 0;
194  size_t hi = *offset_first;
195  exterior = LinearRing(coord_first + lo, coord_first + hi, input_dim);
196  lo = hi;
197  offset_first++;
198 
199  Bounds& b = bounds;
200  Bounds& b_ext = exterior.bounds;
201  b.extend(b_ext.minx, b_ext.miny);
202  b.extend(b_ext.maxx, b_ext.maxy);
203 
205  interiors.reserve((coord_last - coord_first) / ndim);
206  for (auto it = offset_first; it != offset_last; ++it)
207  {
208  hi = *it;
209  interiors.emplace_back(coord_first + lo, coord_first + hi, input_dim);
210  lo = hi;
211  const auto& l = interiors[interiors.size() - 1];
212  bounds.extend(l.bounds.minx, l.bounds.miny);
213  bounds.extend(l.bounds.maxx, l.bounds.maxy);
214  }
215  }

227  static Polygon from_json(const std::string& json)
228  {
229  try
230  {
231  auto j = nlohmann::json::parse(json);
232  auto geom_type = j.at("type").get<std::string>();
233  if (geom_type != "Polygon")
234  {
235  throw exceptions::ParseError("invalid geometry type: " + std::string(geom_type));
236  }
237  const auto& linearrings = j.at("coordinates");
238  std::vector<LinearRing> rings;
239  rings.reserve(linearrings.size());
240  std::vector<Point> points;
241  std::vector<std::size_t> offsets;
242  for (const auto& linearring : linearrings)
243  {
244  if (not linearring.empty())
245  {
246  const auto& coords = linearring.get<std::vector<std::vector<double>>>();
247  points.reserve(coords.size());
248  std::for_each(std::begin(coords), std::end(coords), [&points](const std::vector<double>& coord) {
249  points.emplace_back(coord);
250  });
251  rings.emplace_back(points);
252  points.clear();
253  }
254  }
255  return Polygon(rings);
256  }
257  catch (const nlohmann::json::exception& e)

259  throw exceptions::ParseError("invalid json: " + std::string(e.what()));

261  catch (const exceptions::GeometryError& e)

263  throw exceptions::ParseError("invalid geometry: " + std::string(e.what()));
