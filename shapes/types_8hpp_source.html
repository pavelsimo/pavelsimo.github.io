143 DimensionType get_dim(GeometryDetailedType detailed_type) noexcept
144 {
145  int value = static_cast<int>(detailed_type);
146  if (value >= 1000 and value < 2000)
147  {
148  return DimensionType::XYZ;

151  {
152  return DimensionType::XYM;
153  }
154  if (value >= 3000)

157  }
158  return DimensionType::XY;
