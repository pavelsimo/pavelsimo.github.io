83  basic_multilinestring(CoordIterator coord_first, CoordIterator coord_last, OffsetIterator offset_first, OffsetIterator offset_last)
84  {
85  if (std::distance(coord_first, coord_last) > 0)
86  {
87  auto n = ndim_();
88  this->reserve((coord_last - coord_first) / n);
89  size_t lo = 0;
90  for (auto it = offset_first; it != offset_last; ++it)
91  {
92  size_t hi = *it;
93  this->emplace_back(coord_first + lo, coord_first + hi);
94  lo = hi;
95  }
96  }
97  }

258  static basic_multilinestring<T> from_json_(const std::string& json)
259  {
260  try
261  {
262  auto j = nlohmann::json::parse(json);
263  auto geom_type = j.at("type").get<std::string>();
264  if (geom_type != "MultiLineString")
265  {
266  throw exceptions::parse_error("invalid geometry type: " + std::string(geom_type));
267  }
268  const auto& linestrings = j.at("coordinates");
269  std::vector<T> res;
270  res.reserve(linestrings.size());
271  std::vector<point_type> points;
272  for (const auto& linestring : linestrings)
273  {
274  if (not linestring.empty())
275  {
276  const auto& coords = linestring.get<std::vector<std::vector<double>>>();
277  points.reserve(coords.size());
278  std::for_each(std::begin(coords), std::end(coords),
279  [&points](const std::vector<double>& coord) {
280  points.emplace_back(coord.begin(), coord.end());
281  });
282  res.emplace_back(points.begin(), points.end());
283  }
284  points.clear();
285  }
286  return basic_multilinestring<T>(res.begin(), res.end());
287  }
288  catch (const nlohmann::json::exception& e)
289  {
290  throw exceptions::parse_error("invalid json: " + std::string(e.what()));
291  }
292  catch (const exceptions::geometry_error& e)
293  {
294  throw exceptions::parse_error("invalid geometry: " + std::string(e.what()));
295  }
