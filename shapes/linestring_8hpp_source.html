108  LineString(InputIt first, InputIt last, DimensionType input_dim)
109  {
110  if (std::distance(first, last) > 0)
111  {
112  auto ndim = static_cast<size_t>(utils::get_ndim(input_dim));
113  seq.reserve(std::distance(first, last) / ndim);
114  dim = input_dim;
115  Point p;
116  p.dim = input_dim;
117  for (auto it = first; it != last; it += ndim)
