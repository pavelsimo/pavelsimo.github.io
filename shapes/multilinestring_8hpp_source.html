138  if (geom_type != "MultiLineString")

142  const auto& linestrings = j.at("coordinates");
143  std::vector<LineString> res;
144  res.reserve(linestrings.size());
145  std::vector<Point> points;
146  for (const auto& linestring : linestrings)
147  {
148  if (not linestring.empty())
149  {
150  const auto& coords = linestring.get<std::vector<std::vector<double>>>();
151  points.reserve(coords.size());
152  std::for_each(std::begin(coords), std::end(coords), [&points](const std::vector<double>& coord) {
153  points.emplace_back(coord);

155  res.emplace_back(points);
156  }
157  points.clear();
158  }
