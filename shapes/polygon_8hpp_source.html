207  {

211  auto geom_type = j.at("type").get<std::string>();
212  if (geom_type != "Polygon")
213  {
214  throw exceptions::ParseError("invalid geometry type: " + std::string(geom_type));
215  }
216  const auto& linearrings = j.at("coordinates");
217  std::vector<LinearRing> res;
218  res.reserve(linearrings.size());
219  std::vector<Point> points;
220  for (const auto& linearring : linearrings)
221  {
222  if (not linearring.empty())
223  {
224  const auto& coords = linearring.get<std::vector<std::vector<double>>>();
225  points.reserve(coords.size());
226  std::for_each(std::begin(coords), std::end(coords), [&points](const std::vector<double>& coord) {
227  points.emplace_back(coord);

229  res.emplace_back(points);
230  }
231  points.clear();
232  }
